//References that we get from the main controller script
BlitMaterialFeature analogFeature;
MainGlitchController mainScript;
Material analogMaterial;

//We receive the references needed to change the values of the glitch material
public void OnInit(ScriptableRendererFeature feature, MainGlitchController main)
{
analogFeature = feature as BlitMaterialFeature;
analogMaterial = analogFeature.Material;

mainScript = main;
}

//Update the values on the glitch material on every frame of the transition
public void AnalogGlitchUpdate()
{
float scanlineJitterDisp = mainScript.LerpValues(jitterRange.x, jitterRange.y);
float colorDriftAmount = mainScript.LerpValues(colorDriftRange.x, colorDriftRange.y);
float horizontalShakeAmount = mainScript.LerpValues(horizontalShakeRange.x, horizontalShakeRange.y);

analogMaterial.SetFloat("_ScanLineJitterDisp", scanlineJitterDisp);
analogMaterial.SetFloat("_ColorDriftAmount", colorDriftAmount);
analogMaterial.SetFloat("_HorizontalShake", horizontalShakeAmount);
}
